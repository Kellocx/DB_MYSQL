CREATE TABLE prodotti (
    
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(10) NOT NULL,
    categoria VARCHAR(30) NOT NULL,
    prezzo DECIMAL(10,2) NOT NULL,
    quantita INT NOT NULL DEFAULT 0,
    disponibile BOOLEAN NOT NULL DEFAULT TRUE
    
    );

//seezionare il nome prezzo e quantita dei prodotti con quantita minore di 5

SELECT  nome, prezzo, quantita FROM prodotti WHERE quantita < 5; 


//ordinare tutti i prodotti in ordine per prezzo decrescente
SELECT * FROM prodotti ORDER BY prezzo DESC;

//ordinare tutti i prodotti con un prezzo compreso tra un range di prezzi
SELECT * FROM prodotti WHERE prezzo BETWEEN 100 AND 900;

//ordinare tutti i prodotti in ordine per prezzo crescente
SELECT * FROM prodotti ORDER BY prezzo ASC;

//calcolare il prezzo medio per categoria
SELECT categoria, AVG(prezzo) AS prezzo_medio FROM prodotti GROUP BY categoria;
